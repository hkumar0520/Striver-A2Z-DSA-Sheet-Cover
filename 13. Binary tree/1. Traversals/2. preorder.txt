


class Solution {
public:
    void preOrder(TreeNode* root, vector<int>& ans){
        if(root == nullptr)
            return;

        ans.push_back(root->val);
        preOrder(root->left, ans);
        preOrder(root->right, ans);

        return;
    }
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> ans;
        preOrder(root, ans);
        return ans;
    }
};


TC: O(n)
SC: O(n) + n{
    n = worst case  height of stack space (skewed trees)
}
