// Template of sum<=x is used

class Solution {
  public:
    int smallestSubWithSum(int x, vector<int>& arr) {
        // Your code goes here
        int n = arr.size(), l = 0, r = 0, sum = 0, minLen = INT_MAX;
        
        while(r<n){
            sum += arr[r];
            
            while(sum > x){
                if(sum > x)
                    minLen = min(minLen, (r-l+1));
                
                sum -= arr[l];
                l++;
                
            }
            
            r++;
        }
        if(minLen == INT_MAX)
            return 0;
        return minLen;
    }
};

TC: O(2*n)
SC: O(1)
