class Solution {
  public:
    long long countSubArrayProductLessThanK(const vector<int>& arr, int n,
                                            long long k) 
    {
        long long countProdLessThanK = 0, product = 1;
        int l = 0, r = 0;
        
        while(r < n){
            product  = product * arr[r];
            
            while(product >= k && l<=r){
                product = product / arr[l];
                l++;
            }
            
            if(product < k)
                countProdLessThanK = countProdLessThanK + (r-l+1);
                
            r++;
        }
        return countProdLessThanK;                                       
    }
};


Time Complexity: O(2*n)
sapce Complexity: O(1);
