class Solution{
    public:    
    long long int maxSumWithK(long long int a[], long long int n, long long int k) 
    {
        long long largestSum = LONG_MIN;
        for(int i = 0; i<n; i++){
            long long sum = 0;
            
            for(int j = i; j<n; j++){
                sum += a[j];
                
                if((j-i+1)<k)
                    continue;
                    
                largestSum = max(largestSum, sum);
                
            }
        }
        return largestSum;
    }
};

TC - O(n*n)
