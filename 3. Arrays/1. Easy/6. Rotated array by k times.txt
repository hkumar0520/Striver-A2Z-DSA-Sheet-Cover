

vector<int> rotateArray(vector<int>arr, int k) {
    // Write your code here.
    int n = arr.size();
    vector<int> temp;
    for(int i=0;i<k;i++)
        temp.push_back(arr[i]);

    for(int i=0;i<n-k;i++)
        arr[i] = arr[i+k];
    
    for(int i=0;i<k;i++)
        arr[n-k+i] = temp[i];

    return arr;
}
