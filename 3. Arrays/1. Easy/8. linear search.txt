

class Solution {
  public:
    // Function to find element in sorted array
    // arr: input array
    // k: element to be searched
    bool searchInSorted(vector<int>& arr, int k) {

        // Your code here
        for(int num: arr){
            if(num == k)    return true;
        }
        return false;
    }
};

TC: O(n)
Sc: O(1)
