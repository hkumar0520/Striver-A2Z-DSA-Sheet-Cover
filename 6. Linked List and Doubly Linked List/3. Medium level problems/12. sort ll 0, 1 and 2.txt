




/*  Node is defined as
  struct Node {
    int data;
    struct Node *next;
    Node(int x) {
        data = x;
        next = NULL;
    }
};*/
class Solution {
  public:
  Node* partition(struct Node* head, int x) {
        // code here
        Node* lessThanX = new Node(-1);
        Node* equalToX = new Node(-1);
        Node* greaterThanX = new Node(-1);
        
        Node* lessThanXHead = lessThanX,
              *equalToXHead = equalToX, 
              *greaterThanXHead = greaterThanX;
        
        Node* current = head;
        while(current != nullptr){
            if(current->data < x){
                lessThanX->next = current;
                lessThanX = lessThanX->next;
            }
            else if(current->data == x){
                equalToX->next = current;
                equalToX = equalToX->next;
            }
            else{
                greaterThanX->next = current;
                greaterThanX = greaterThanX->next;
            }
            
            current = current->next;
        }
        
        greaterThanX->next = nullptr;
        equalToX->next = greaterThanXHead->next;
        lessThanX->next = equalToXHead->next;
        
        return lessThanXHead->next;
    }
    Node* segregate(Node* head) {
        // code here
        return partition(head, 1);
    }
};


TC: O(n)
SC: O(1)
