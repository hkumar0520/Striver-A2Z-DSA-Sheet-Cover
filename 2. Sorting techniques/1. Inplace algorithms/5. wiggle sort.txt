#include <bits/stdc++.h> 
vector<int> wiggleSort(int n, vector<int> nums) {
    // Write your code here

    for(int i = 0; i<(n-1); i++){
        if( i%2 == 0){
            if(nums[i] > nums[i+1])
                swap(nums[i], nums[i+1]);
        }
        else{
            if(nums[i] < nums[i+1])
                swap(nums[i], nums[i+1]);
        }
    }
    return nums;
}

TC: O(n)
SC: O(1)

    for(int i = 1; i<n; i+= 2){
        int left = arr[i-1], right = ((i+1) < n)? arr[i+1] : INT_MIN;
        if( (left > right) && (left>arr[i]) )
            swap(arr[i], arr[i-1]);
        else if( (right > left) && (right > arr[i])){
            if(i+1 < n)
                swap(arr[i], arr[i+1]);
        }
    }
    return arr;
}
