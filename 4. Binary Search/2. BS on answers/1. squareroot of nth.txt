

int floorSqrt(int n)
{
  if(n==0 || n==1)  return n;
  int i = 1, j = n-1;
  while(i<=j)
  {
    int mid = (i+j)/2;
    // sometimes val is out of bounds, then its value -2147483648 subsituted
    int val = pow(mid,2); 
    if(val == n)
      return mid;
    else if(val > n || (val == -2147483648 )) // in case val is outof bounds
      j = mid-1;
    else
      i = mid+1;
  }
  // return floor if sqrt not available
  return j;
}
