
int count(vector<int>& arr, int n, int x) {
	// Write Your Code Here
	int firstt = -1, lastt = -1, i = 0, j = n-1;
	while(i<=j)
	{
		int mid = (i+j)/2;
		if(arr[mid]>= x)
		{
			if(arr[mid] == x)
				firstt = mid;
			j = mid-1;
		}
		else
			i = mid+1;
	}
	if(firstt == -1)	return 0;
	i = 0, j=n-1;
	while(i<=j)
	{
		int mid = (i+j)/2;
		if(arr[mid]<= x)
		{
			if(arr[mid] == x)
				lastt = mid;
			i = mid+1;
		}
		else
			j = mid-1;
	}
	return (lastt - firstt +1);
}
