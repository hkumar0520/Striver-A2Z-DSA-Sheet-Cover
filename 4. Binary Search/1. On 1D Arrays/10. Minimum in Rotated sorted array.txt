
int findMin(vector<int>& arr)
{
	// Write your code here.
	int n = arr.size();
	int i=0,j = n-1, ans = INT_MAX;
    while(i<=j)
    {
        int mid = (i+j)/2;

        if(arr[i]<=arr[j])
        {
            ans = min(ans,arr[i]);
            break;
        }

        if(arr[i]<=arr[mid])
        {
            ans = min(ans,arr[i]);
            i = mid+1;
        }
        else
        {
            ans = min(ans,arr[mid]);
            j = mid-1;
        }
        
    }
	return ans;
}
