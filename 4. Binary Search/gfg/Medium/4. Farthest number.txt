class Solution{   
  public:
    vector<int> farNumber(int N,vector<int> Arr){
        //code here
        vector<int> ans(N, -1);
        for(int i = 0; i<N; i++){
            for(int j = N-1; j>i; j--){
                if( Arr[i] > Arr[j]){
                    ans[i] = j;
                    break;
                }
            }
        }
        return ans;
    }
    
};

TC - O(N*N)
SC - O(1)
