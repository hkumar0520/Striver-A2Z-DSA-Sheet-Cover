/************************************************************

    Following is the TreeNode class structure

    template <typename T>
    class TreeNode {
       public:
        T val;
        TreeNode<T> *left;
        TreeNode<T> *right;

        TreeNode(T val) {
            this->val = val;
            left = NULL;
            right = NULL;
        }
    };

************************************************************/

int floorInBST(TreeNode<int> * root, int X)
{
    // Write your code here.
    if(root == nullptr) return -1;
    if(root->val == X)  return root->val;

    if(root->val < X)
        return max( root->val, floorInBST(root->right, X));
    else
        return floorInBST(root->left, X);
    
}


TC: O(logn)
SC: O(logn)

#include <bits/stdc++.h> 


int floorInBST(TreeNode<int> * root, int X)
{
    // Write your code here.
    TreeNode<int>* currentNode = root, *floor = nullptr;

    while(currentNode != nullptr){
        
        if(currentNode->val == X){
            floor = currentNode;
            break;
        }
        else if(currentNode->val < X){
            floor = currentNode;
            currentNode = currentNode->right;
        }
        else
            currentNode = currentNode->left;
    }

    return floor == nullptr ? -1 : floor->val;
}


TC: O( log(n))
SC: O(1)
