//Function to extract minimum value in heap and then to store 
//next minimum value at first index.
int MinHeap::extractMin() 
{
    // Your code here
    if(heap_size == 0)      return -1;
    
    int minElement = harr[0];
    harr[0] = harr[heap_size-1];
    heap_size--;
    MinHeapify(0);
    
    return minElement;
}

//Function to insert a value in Heap.
void MinHeap::insertKey(int k) 
{
    if(heap_size == capacity)   return;
  
    harr[heap_size] = k;
    heap_size++;
    int i = heap_size-1;
    while ( i != 0 && harr[parent(i)] > harr[i]){
        swap(harr[i], harr[parent(i)]);
        i = parent(i);
    }
}

//Function to change value at ith index and store that value at first index.
void MinHeap::decreaseKey(int i, int new_val) 
{
    harr[i] = new_val;
    while (i != 0 && harr[parent(i)] > harr[i]) {
        swap(harr[i], harr[parent(i)]);
        i = parent(i);
    }
}

void MinHeap::MinHeapify(int i) 
{
    int l = left(i);
    int r = right(i);
    int smallest = i;
    if (l < heap_size && harr[l] < harr[i]) smallest = l;
    if (r < heap_size && harr[r] < harr[smallest]) smallest = r;
    if (smallest != i) {
        swap(harr[i], harr[smallest]);
        MinHeapify(smallest);
    }
}

#include <iostream>
#include <vector>
#include <climits>
using namespace std;

class MinHeap {
private:
    vector<int> heap;

    // Helper function to maintain the heap property after insertion
    void heapifyUp(int index) {
        while (index > 0) {
            int parent = (index - 1) / 2;
            if (heap[index] < heap[parent]) {
                swap(heap[index], heap[parent]);
                index = parent;
            } else {
                break;
            }
        }
    }

    // Helper function to maintain the heap property after extraction
    void heapifyDown(int index) {
        int left = 2 * index + 1;
        int right = 2 * index + 2;
        int smallest = index;

        if (left < heap.size() && heap[left] < heap[smallest]) {
            smallest = left;
        }

        if (right < heap.size() && heap[right] < heap[smallest]) {
            smallest = right;
        }

        if (smallest != index) {
            swap(heap[index], heap[smallest]);
            heapifyDown(smallest);
        }
    }

public:
    // Insert a new element into the heap
    void insert(int value) {
        heap.push_back(value);
        heapifyUp(heap.size() - 1);
    }

    // Extract the minimum element (root)
    int extractMin() {
        if (heap.empty()) {
            cout << "Heap is empty!" << endl;
            return INT_MAX;
        }
        int minValue = heap[0];
        heap[0] = heap.back();
        heap.pop_back();
        heapifyDown(0);
        return minValue;
    }

    // Peek the minimum element (root)
    int peekMin() const {
        if (heap.empty()) {
            cout << "Heap is empty!" << endl;
            return INT_MAX;
        }
        return heap[0];
    }

    // Build the heap from an unordered array
    void buildHeap(const vector<int>& arr) {
        heap = arr;
        for (int i = (heap.size() / 2) - 1; i >= 0; i--) {
            heapifyDown(i);
        }
    }

    // Get the size of the heap
    int size() const {
        return heap.size();
    }

    // Print the heap for debugging
    void printHeap() const {
        for (int i = 0; i < heap.size(); i++) {
            cout << heap[i] << " ";
        }
        cout << endl;
    }
};

int main() {
    MinHeap heap;

    // Inserting elements
    heap.insert(10);
    heap.insert(20);
    heap.insert(5);
    heap.insert(30);
    heap.insert(15);
    
    cout << "Heap after insertions: ";
    heap.printHeap();

    // Extracting the minimum element
    cout << "Extracted min: " << heap.extractMin() << endl;
    
    cout << "Heap after extracting min: ";
    heap.printHeap();

    // Peek the minimum element
    cout << "Peek min: " << heap.peekMin() << endl;

    // Build a heap from an unordered array
    vector<int> arr = {9, 5, 7, 3, 10, 1};
    heap.buildHeap(arr);
    
    cout << "Heap after building from array: ";
    heap.printHeap();

    return 0;
}

#include <iostream>
#include <vector>
#include <climits>
using namespace std;

class MinHeap {
private:
    vector<int> heap;

    // Helper function to maintain the heap property for downward adjustment
    void heapifyDown(int index) {
        int smallest = index; // Initialize smallest as root
        int left = 2 * index + 1; // Left child
        int right = 2 * index + 2; // Right child

        // If left child is smaller than root
        if (left < heap.size() && heap[left] < heap[smallest])
            smallest = left;

        // If right child is smaller than smallest so far
        if (right < heap.size() && heap[right] < heap[smallest])
            smallest = right;

        // If smallest is not root
        if (smallest != index) {
            swap(heap[index], heap[smallest]); // Swap
            heapifyDown(smallest); // Recursively heapify the affected subtree
        }
    }

    // Helper function to maintain the heap property for upward adjustment
    void heapifyUp(int index) {
        int parent = (index - 1) / 2;

        // While the current node is smaller than its parent
        if (index > 0 && heap[index] < heap[parent]) {
            swap(heap[index], heap[parent]); // Swap
            heapifyUp(parent); // Recursively heapify the parent node
        }
    }

public:
    // Insert a new key into the heap
    void insert(int key) {
        heap.push_back(key); // Add the key at the end
        heapifyUp(heap.size() - 1); // Fix the heap property
    }

    // Remove and return the minimum element (root)
    int extractMin() {
        if (heap.empty()) {
            cout << "Heap is empty!" << endl;
            return -1;
        }

        int root = heap[0];
        heap[0] = heap.back(); // Move the last element to the root
        heap.pop_back(); // Remove the last element
        heapifyDown(0); // Fix the heap property
        return root;
    }

    // Get the minimum element (root) without removing it
    int getMin() {
        if (heap.empty()) {
            cout << "Heap is empty!" << endl;
            return -1;
        }
        return heap[0];
    }

    // Delete a specific key from the heap
    void deleteKey(int index) {
        if (index < 0 || index >= heap.size()) {
            cout << "Index out of bounds!" << endl;
            return;
        }

        // Reduce the value to a very small number and move it to the root
        decreaseKey(index, INT_MIN);
        extractMin(); // Remove the root
    }

    // Decrease the value of a specific key
    void decreaseKey(int index, int newValue) {
        if (index < 0 || index >= heap.size()) {
            cout << "Index out of bounds!" << endl;
            return;
        }

        if (newValue > heap[index]) {
            cout << "New value is greater than the current value!" << endl;
            return;
        }

        heap[index] = newValue; // Update the value
        heapifyUp(index); // Fix the heap property
    }

    // Replace the root with a new value
    void replaceRoot(int key) {
        if (heap.empty()) {
            cout << "Heap is empty!" << endl;
            return;
        }

        heap[0] = key;
        heapifyDown(0);
    }

    // Print the heap elements
    void printHeap() {
        for (int val : heap)
            cout << val << " ";
        cout << endl;
    }

    // Get the size of the heap
    int size() {
        return heap.size();
    }

    // Check if the heap is empty
    bool isEmpty() {
        return heap.empty();
    }
};

int main() {
    MinHeap minHeap;

    // Insert elements into the heap
    minHeap.insert(10);
    minHeap.insert(20);
    minHeap.insert(5);
    minHeap.insert(15);
    minHeap.insert(30);

    cout << "Heap elements: ";
    minHeap.printHeap();

    cout << "Minimum element: " << minHeap.getMin() << endl;

    cout << "Extracted minimum: " << minHeap.extractMin() << endl;

    cout << "Heap elements after extraction: ";
    minHeap.printHeap();

    cout << "Decreasing value at index 2 to 1" << endl;
    minHeap.decreaseKey(2, 1);

    cout << "Heap elements after decreasing key: ";
    minHeap.printHeap();

    cout << "Deleting element at index 1" << endl;
    minHeap.deleteKey(1);

    cout << "Heap elements after deletion: ";
    minHeap.printHeap();

    cout << "Replacing root with 25" << endl;
    minHeap.replaceRoot(25);

    cout << "Heap elements after replacing root: ";
    minHeap.printHeap();

    cout << "Heap size: " << minHeap.size() << endl;
    cout << "Is heap empty? " << (minHeap.isEmpty() ? "Yes" : "No") << endl;

    return 0;
}





Here is the detailed analysis of the time and space complexity for each operation in the provided **Min-Heap** implementation:

---

### **1. `insert(int key)`**
- **Operation**: Adds a new element to the heap and restores the heap property by calling `heapifyUp`.
- **Time Complexity**: 
  - `heapifyUp` is called, which may need to traverse up to the root of the heap.
  - The maximum number of levels in the heap is \( \log n \), where \( n \) is the number of elements in the heap.
  - Hence, the complexity is \( O(\log n) \).
- **Space Complexity**: 
  - No additional space is used apart from the input key.
  - **Space**: \( O(1) \).

---

### **2. `extractMin()`**
- **Operation**: Removes the root (minimum element), replaces it with the last element, and restores the heap property by calling `heapifyDown`.
- **Time Complexity**:
  - `heapifyDown` is called, which may need to traverse down to the leaf level.
  - The maximum number of levels in the heap is \( \log n \).
  - Hence, the complexity is \( O(\log n) \).
- **Space Complexity**: 
  - No additional space is used.
  - **Space**: \( O(1) \).

---

### **3. `getMin()`**
- **Operation**: Returns the root (minimum element) without modifying the heap.
- **Time Complexity**:
  - Accessing the root is \( O(1) \).
- **Space Complexity**: 
  - No additional space is used.
  - **Space**: \( O(1) \).

---

### **4. `deleteKey(int index)`**
