


// User function Template for C++

class Solution {
  public:
    void clonestack(stack<int> st, stack<int>& cloned) {
        // code here
        if(st.empty())
            return ;
            
        int topElement = st.top();
        st.pop();
        
        clonestack(st, cloned);
        cloned.push(topElement);
    }
};

TC: O(n)
SC: O(n) {
    stack space 
}
