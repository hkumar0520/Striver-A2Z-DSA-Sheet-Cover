void MyStack ::push(int x) {
    // Your Code
    if(top + 1 == 1000) return ;
    
    top++;
    arr[top] = x;
}

// Function to remove an item from top of the stack.
int MyStack ::pop() {
    // Your Code
    if(top == -1)
        return -1;
        
    int topElement = arr[top];
    top--;
    return topElement;
}


TC: O(1)
SC: O(n)
