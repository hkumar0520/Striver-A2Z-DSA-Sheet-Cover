
#include <bits/stdc++.h> 


int maximumProfit(vector<int> &prices){
    // Write your code here.
    int mini = prices[0],result = INT_MIN;
    int n = prices.size();
    for(int i=0;i<n;i++)
    {
        mini = min(mini, prices[i]);
        result = max(result, prices[i]-mini);
    }
    return result;
}
