
// recursion 
int f(int ind, int buy, vector<int> &values, int n, int fee)
{
    if(ind == n)    return 0;

    int opt1 = 0, opt2 = 0;
    if(buy == 0)
    {
        opt1 = f(ind+1,buy,values,n, fee);
        opt2 = -values[ind] + f(ind+1,1,values,n, fee);
    }
    else
    {
        opt1 = f(ind+1,buy,values,n, fee);
        opt2 = values[ind]-fee + f(ind+1,0,values,n, fee);
    }
    return max(opt1,opt2);
}

int maximumProfit(vector<int> &prices, int n, int fee)
{
	// Write your code here.
	return f(0, 0, prices, n, fee);

}

// memoization

#include <bits/stdc++.h>

int f(int ind, int buy, vector<int> &values, int n, vector<vector<int>> &dp, int fee)
{
    
    if(ind == n)    return 0;

    if(dp[ind][buy] != -1)   return dp[ind][buy];

    int opt1 = 0, opt2 = 0;
    if(buy == 0)
    {
        opt1 = f(ind+1,buy,values,n,dp, fee);
        opt2 = -values[ind] + f(ind+1,1,values,n,dp, fee);
    }
    else
    {
        opt1 = f(ind+1,buy,values,n,dp, fee);
        opt2 = values[ind] - fee + f(ind+1,0,values,n,dp, fee);
    }
    return dp[ind][buy] = max(opt1,opt2);
}

int maximumProfit(vector<int> &prices, int n, int fee)
{
	// Write your code here.
	vector<vector<int>> dp(n,vector<int>(2,-1));
	return f(0, 0, prices, n,dp, fee);

}

// tabular

class Solution {
public:
    int f(vector<int> prices, int fee, vector<vector<int>>& dp)
    {
        int n = prices.size();
        for(int i = n-1;i>=0;i--)
        {
            for(int buy = 0;buy<2;buy++)
            {
                if(buy == 0)
                {
                    dp[i][buy] = max(dp[i+1][buy],
                                    -prices[i]+dp[i+1][1]);
                }
                else
                {
                    dp[i][buy] = max(dp[i+1][buy],
                                    +prices[i]-fee+dp[i+1][0]);
                }
            }
        }
        return dp[0][0];
    }
    int maxProfit(vector<int>& prices, int fee) {
        int  n = prices.size();
        vector<vector<int>> dp(n+1,vector<int>(2,0));
        return f(prices,fee,dp);
    }
};

// optimization

class Solution {
public:
    int f(vector<int> prices, int fee)
    {
        vector<int> curr(2,0);
        vector<int> ahead(2,0);
        int n = prices.size();
        for(int i = n-1;i>=0;i--)
        {
            for(int buy = 0;buy<2;buy++)
            {
                if(buy == 0)
                {
                    curr[buy] = max(ahead[buy],
                                    -prices[i]+ahead[1]);
                }
                else
                {
                    curr[buy] = max(ahead[buy],
                                    +prices[i]-fee+ahead[0]);
                }
            }
            ahead = curr;
        }
        return curr[0];
    }
    int maxProfit(vector<int>& prices, int fee) {
        int  n = prices.size();
        return f(prices,fee);
    }
};
