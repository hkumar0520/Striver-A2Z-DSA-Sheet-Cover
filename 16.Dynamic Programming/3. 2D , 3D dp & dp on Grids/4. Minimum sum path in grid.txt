#include <bits/stdc++.h> 
int minSumPath(vector<vector<int>> &grid) {
    // Write your code here.
    int n = grid.size();
    int m = grid[0].size();
    vector<int> dp(m,INT_MAX);
    dp[0] = 0;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            if(j>0)
                dp[j] = grid[i][j] + min(dp[j], dp[j-1]);
            else
                dp[j] = grid[i][j] + dp[j];
        }
    }
    return dp[m-1];
}
