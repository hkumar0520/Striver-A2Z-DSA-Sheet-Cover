#include <bits/stdc++.h>

int ninjaTraining(int n, vector<vector<int>> &points)
{
    vector<vector<int>> dp(n, vector<int>(4, -1));
    // Write your code here.

    dp[0][0] = max(points[0][1], points[0][2]);
    dp[0][1] = max(points[0][0], points[0][2]);
    dp[0][2] = max(points[0][0], points[0][1]);

    dp[0][3] = max({ dp[0][0], dp[0][1], dp[0][2]});

    for(int day = 1; day<n; day++){
        for(int last = 0; last<4; last++){
            dp[day][last] = 0;
            for(int task = 0; task<3; task++){
                if(last == task)    continue;

                int pointDay = points[day][task] + dp[day-1][task];
                dp[day][last] = max(dp[day][last], pointDay);
                
            }
        }
    }

    return dp[n-1][3];
}

TC: n*4*3  = n
SC: O(n*4)

#include <bits/stdc++.h>

int ninjaTraining(int n, vector<vector<int>> &points)
{
    vector<int> prev(4, -1);
    // Write your code here.

    prev[0] = max(points[0][1], points[0][2]);
    prev[1] = max(points[0][0], points[0][2]);
    prev[2] = max(points[0][0], points[0][1]);

    prev[3] = max({ prev[0], prev[1], prev[2]});

    for(int day = 1; day<n; day++){
        vector<int> current = prev;
        for(int last = 0; last<4; last++){
            for(int task = 0; task<3; task++){
                if(last == task)    continue;

                int pointDay = points[day][task] + prev[task];
                current[last] = max(current[last], pointDay);
                
            }
        }
        prev = current;
    }

    return prev[3];
}

TC: O(n*4*3) : O(n)
SC: O(1)
