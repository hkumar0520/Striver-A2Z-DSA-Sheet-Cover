#include<bits/stdc++.h>
using namespace std;


int f(int n, vector<int>& dp)
{
        if(n<=1)        return n;

        if(dp[n-1] != -1)       return dp[n-1];

        dp[n-1] = f(n-1,dp) + f(n-2,dp);

        return dp[n-1];
}


int main()
{

        int n;

        vector<int> dp(n,-1);
        int output  = f(n,dp);
        cout<<output<<endl;
        return 0;
}

#include<bits/stdc++.h>
using namespace std;


int f(int n, vector<int>& dp)
{
        dp[0] = 0;
        dp[1] = 1;
        for(int i=2;i<= n;i++)
                dp[i] = dp[i-1]+dp[i-2];

        return dp[n];
}

#include<bits/stdc++.h>
using namespace std;


int f(int n)
{
        int prev = 0, curr = 1;
        for(int i=2;i<= n;i++)
        {
                int nextt = prev + curr;
                prev = curr;
                curr = nextt;
        }

        return curr;
}

class Solution {
  public:
    int f(int n, vector<int>& dp){
        // code here
        if(n<=1)    return n;
        
        return dp[n] = f(n-1, dp) + f(n-2, dp);
        
    }
    int nthFibonacci(int n) {
        vector<int> dp(n+1, -1);
        return f(n, dp);
    }
};

TC: O(n)
SC: O(n) + O(n){
        n = stack space
        n = dp space
}

// optimization

class Solution {
  public:
    int f(int n){
       if(n <= 1)   return n;
        // code here
       int prev = 1, prevTwo = 0;
       for(int i = 2; i<=n; i++){
           int curr = prev + prevTwo;
           prevTwo = prev;
           prev = curr;
       }
       return prev;
    }
    int nthFibonacci(int n) {
        
        return f(n);
    }
};

TC: O(n)
SC: O(1)
