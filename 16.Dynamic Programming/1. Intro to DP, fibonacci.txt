#include<bits/stdc++.h>
using namespace std;


int f(int n, vector<int>& dp)
{
        if(n<=1)        return n;

        if(dp[n-1] != -1)       return dp[n-1];

        dp[n-1] = f(n-1,dp) + f(n-2,dp);

        return dp[n-1];
}


int main()
{
        /*
        *  Write your code here. 
         *  Read input as specified in the question.
         *  Print output as specified in the question.
        */
        int n;
        cin>>n;
        vector<int> dp(n,-1);
        int output  = f(n,dp);
        cout<<output<<endl;
        return 0;
}

int f(int n, vector<int>& dp)
{
        dp[0] = 0;
        dp[1] = 1;
        for(int i=2;i<= n;i++)
                dp[i] = dp[i-1]+dp[i-2];

        return dp[n];
}

#include<bits/stdc++.h>
using namespace std;


int f(int n)
{
        int prev = 0, curr = 1;
        for(int i=2;i<= n;i++)
        {
                int nextt = prev + curr;
                prev = curr;
                curr = nextt;
        }

        return curr;
}


int main()
{
        /*
        *  Write your code here. 
         *  Read input as specified in the question.
         *  Print output as specified in the question.
        */
        int n;

        int output  = f(n);

        return 0;
}
