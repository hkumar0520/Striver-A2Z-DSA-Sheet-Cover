class Solution {
  public:
    int countNumberswith4(int n) {
        // code here
        int countNumbers = 0;
        for(int i = 4; i<=n; i++){
            string str = to_string(i);
            if(to_string(i).find("4") != std::string::npos)
                countNumbers++;
        }   
        return countNumbers;
    }
};

TC: O(n)
SC: O(1)
